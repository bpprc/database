162 def accession_availability(self, obj):

164 if not accession: 

165 return format_html('<body> <p>No accession number</p> </body>') 

166 try: 

167 handle = Entrez.efetch( 

168 db="nucleotide", id=accession, rettype="fasta", retmode="text") 

169 return format_html('<body> <p style="color:#FF0000";>Sequence is Public</p> </body>') 

170 except:
