52def guess_if_protein(seq, thresh=0.99): 

53 """Guess if the given sequence is Protein.""" 

54 # protein_letters = ['C', 'D', 'S', 'Q', 'K','I','P','T','F','N','G', 

55 # 'H','L','R','W','A','V','E','Y','M'] 

56 dna_letters = ['A', 'C', 'G', 'T'] 

57 # print(seq) 

58 

59 for record in SeqIO.parse(seq, "fasta"): 

60 seq = record.seq 

61 

62 seq = seq.upper() 

63 protein_alpha_count = 0 

64 for letter in dna_letters: 

65 protein_alpha_count += seq.count(letter) 

66 

67 return (len(seq) == 0 or float(protein_alpha_count) / float(len(seq)) >= thresh)

336 #invalidsymbols = invalid_symbol(sequence_in_form)

342 raise forms.ValidationError( 

343 "There are numbers in the sequence. Please paste protein sequence only") 

344 

345 if sequence_in_form:
